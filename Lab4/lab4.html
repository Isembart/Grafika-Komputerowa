def RPYT(roll, pitch, yaw, px, py, pz):
    r = np.deg2rad(roll)
    p = np.deg2rad(pitch)
    y = np.deg2rad(yaw)

    MR = np.array(
        [
            [np.cos(r), -np.sin(r), 0.0],
            [np.sin(r), np.cos(r), 0.0],
            [0.0, 0.0, 1.0],
        ],
        dtype=float,
    )

    MP = np.array(
        [
            [np.cos(p), 0.0, np.sin(p)],
            [0.0, 1.0, 0.0],
            [-np.sin(p), 0.0, np.cos(p)],
        ],
        dtype=float,
    )

    MY = np.array(
        [
            [1.0, 0.0, 0.0],
            [0.0, np.cos(y), -np.sin(y)],
            [0.0, np.sin(y), np.cos(y)],
        ],
        dtype=float,
    )

    Rot = MY @ MP @ MR

    Hmat = np.zeros((4, 4), dtype=float)
    Hmat[:3, :3] = Rot
    Hmat[0, 3] = px
    Hmat[1, 3] = py
    Hmat[2, 3] = pz
    Hmat[3, 3] = 1.0

    return Hmat


def linia1(Im1, yp, xp, yk, xk, C):
    Y, X = Im1.shape
    Im2 = Im1.copy()

    if abs(yk - yp) > abs(xk - xp):
        x0, y0, x1, y1 = yp, xp, yk, xk
        zamiana = 1
    else:
        x0, y0, x1, y1 = xp, yp, xk, yk
        zamiana = 0

    if zamiana == 1:
        X, Y = Y, X

    if x0 > x1:
        x0, x1 = x1, x0
        y0, y1 = y1, y0

    dx = abs(x1 - x0)
    dy = abs(y1 - y0)
    sy = 1 if y1 > y0 else (-1 if y1 < y0 else 0)

    x, y = x0, y0

    def draw_point(px, py):
        if 0 <= px < X and 0 <= py < Y:
            if zamiana == 0:
                Im2[py, px] = C
            else:
                Im2[px, py] = C

    draw_point(x, y)

    error = 2 * dy - dx
    for _ in range(dx):
        error += 2 * dy
        if error > 0:
            y += sy
            error -= 2 * dx
        x += 1
        draw_point(x, y)

    return Im2


def Persp(f):
    H = np.zeros((4, 4), dtype=float)
    H[0, 0] = 1.0
    H[1, 1] = 1.0
    H[3, 2] = -1.0 / f
    H[3, 3] = 1.0
    return H

Przekazywanie T¶

Jaki sens ma przekazywanie koloru początkowego T skoro przekazujemy współrzędne startowe? Skoro mamy współrzędne startowe, to możemy odczytać kolor początkowy z obrazu, który użytkownik chcę zamienić. Przekazanie T wymaga wcześniejszego odczytania koloru współrzędnych startowych przed wywołaniem funkcji. Ten mechanizm jest według mnie zbędny i nieprawidłowy z punktu implementacyjnego więc decyduje się go pominąć.

def floodfill(Image, x, y, newColor):
    imgCopy = Image.copy()

    def floodfill_inplace(Image, x, y, newColor):
        oldColor = Image[y][x].copy()
        if (oldColor == newColor).all():
            return
        Image[y][x] = newColor
        if x > 0 and (Image[y][x - 1] == oldColor).all():
            floodfill_inplace(Image, x - 1, y, newColor)
        if x < len(Image[0]) - 1 and (Image[y][x + 1] == oldColor).all():
            floodfill_inplace(Image, x + 1, y, newColor)
        if y > 0 and (Image[y - 1][x] == oldColor).all():
            floodfill_inplace(Image, x, y - 1, newColor)
        if y < len(Image) - 1 and (Image[y + 1][x] == oldColor).all():
            floodfill_inplace(Image, x, y + 1, newColor)

    floodfill_inplace(imgCopy, x, y, newColor)
    return imgCopy

def render_frame(roll, pitch, yaw, px, py, pz):
    frame = np.full((400, 400), 10, dtype=np.uint8)

    t1 = np.array([[0, 0, 0], [1, 0, 0], [0, 1, 0]])

    H = RPYT(roll, pitch, yaw, px, py, pz)
    t_world = t1 @ H[:3, :3].T + H[:3, 3]

    HHH = Persp(1.0)
    t_proj = t_world @ HHH[:3, :3].T + HHH[:3, 3]
    x_proj = t_proj[:, 0] * 100
    y_proj = t_proj[:, 1] * 100

    # Krawedzie trojkata
    for i in range(3):
        x0, y0 = int(x_proj[i]), int(y_proj[i])
        x1, y1 = int(x_proj[(i + 1) % 3]), int(y_proj[(i + 1) % 3])
        frame = linia1(frame, y0, x0, y1, x1, 255)

    cx = int(np.mean(x_proj))
    cy = int(np.mean(y_proj))
    if 0 <= cx < 400 and 0 <= cy < 400:
        frame = floodfill(frame, cx, cy, 255)

    return frame

def imageFloodFill(Image, x, y, newImage):
    imgCopy = Image.copy()

    def floodfill_inplace(Image, x, y, newImage, newImageX, newImageY):
        oldColor = Image[y][x].copy()
        if (oldColor == newImage[newImageY][newImageX]).all():
            return
        Image[y][x] = newImage[newImageY][newImageX].copy()

        if x > 0 and (Image[y][x - 1] == oldColor).all():
            floodfill_inplace(
                Image,
                x - 1,
                y,
                newImage,
                (newImageX - 1) % newImage.shape[1],
                (newImageY) % newImage.shape[0],
            )
        if x < len(Image[0]) - 1 and (Image[y][x + 1] == oldColor).all():
            floodfill_inplace(
                Image,
                x + 1,
                y,
                newImage,
                (newImageX + 1) % newImage.shape[1],
                (newImageY) % newImage.shape[0],
            )
        if y > 0 and (Image[y - 1][x] == oldColor).all():
            floodfill_inplace(
                Image,
                x,
                y - 1,
                newImage,
                (newImageX) % newImage.shape[1],
                (newImageY - 1) % newImage.shape[0],
            )
        if y < len(Image) - 1 and (Image[y + 1][x] == oldColor).all():
            floodfill_inplace(
                Image,
                x,
                y + 1,
                newImage,
                (newImageX) % newImage.shape[1],
                (newImageY + 1) % newImage.shape[0],
            )

    floodfill_inplace(imgCopy, x, y, newImage, 1, 1)
    return imgCopy
