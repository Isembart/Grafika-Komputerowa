def sqrt_bres(a1, m, n):
    im = np.full((m, n), 255, dtype=np.uint8)
    a = int(a1)
    asq = a * a
    y_switch = asq // 2
    if y_switch > m - 1:
        y_switch = m - 1

    x = 0
    for y in range(y_switch + 1):
        xt = y * y
        if abs(xt - asq * x) > abs(xt - asq * (x + 1)):
            x += 1
        if x < n:
            im[m - y - 1, x] = 0

    xp = x
    y = y_switch
    for x in range(xp, n):
        ytsq = asq * x
        if abs(ytsq - y * y) > abs(ytsq - (y + 1) * (y + 1)):
            y += 1
        if y > m - 1:
            break
        im[m - y - 1, x] = 0

    return im

def MSElin(points, functionParams):
    # m = liczba punktów
    V = np.zeros((len(points), len(functionParams)))

    for i in range(len(points)):
        for idx, param in enumerate(functionParams):
            V[i][idx] = param(points[i][0])

    beta = np.linalg.inv(V.T @ V) @ V.T @ points[:, 1]
    return beta
